    <script>
        // Enhanced tracking for commercial security
        function trackCommercialSecurityRequest() {
            gtag('event', 'generate_lead', {
                'event_category': 'Commercial Security Services',
                'event_label': 'commercial_security',
                'value': 1,
                'custom_parameter_1': 'commercial_security',
                'business_size': 'all_sizes'
            });
            
            fbq('track', 'Lead', {
                content_name: 'Commercial Security Assessment',
                content_category: 'Commercial Security Services',
                value: <%= serviceData.priceRange.low %>,
                currency: 'USD',
                custom_data: {
                    'security_type': 'commercial',
                    'certification': 'ISO_27001'
                }
            });
            
            // Redirect to business assessment page
            window.location.href = '/business-security-assessment';
        }

        // Smooth scrolling for anchor links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({
                        behavior: 'smooth',
                        block: 'start'
                    });
                }
            });
        });

        // Intersection Observer for animations
        const observerOptions = {
            threshold: 0.1,
            rootMargin: '0px 0px -100px 0px'
        };

        const observer = new IntersectionObserver(function(entries) {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.style.opacity = '1';
                    entry.target.style.transform = 'translateY(0)';
                    entry.target.classList.add('animate-on-scroll');
                }
            });
        }, observerOptions);

        // Observe elements for animation
        document.querySelectorAll('.commercial-card, .service-card, .process-step, .benefit-item, .industry-card, .package-card').forEach(el => {
            el.style.opacity = '0';
            el.style.transform = 'translateY(30px)';
            el.style.transition = 'all 0.6s ease';
            observer.observe(el);
        });

        // Counter animation for statistics
        function animateCounter(element, target) {
            let current = 0;
            const increment = target / 100;
            const timer = setInterval(() => {
                current += increment;
                if (current >= target) {
                    current = target;
                    clearInterval(timer);
                }
                const displayValue = Math.floor(current);
                const originalText = element.textContent;
                if (originalText.includes('+')) {
                    element.textContent = displayValue + '+';
                } else if (originalText.includes('%')) {
                    element.textContent = displayValue + '%';
                } else if (originalText.includes('/')) {
                    element.textContent = originalText;
                } else {
                    element.textContent = displayValue;
                }
            }, 20);
        }

        // Trigger counter animation when stats come into view
        const statObserver = new IntersectionObserver(function(entries) {
            entries.forEach(entry => {
                if (entry.isIntersecting && !entry.target.classList.contains('animated')) {
                    entry.target.classList.add('animated');
                    const statNumber = entry.target.querySelector('.stat-number');
                    const value = parseInt(statNumber.textContent.replace(/[^\d]/g, ''));
                    if (value && value > 0) {
                        animateCounter(statNumber, value);
                    }
                }
            });
        }, { threshold: 0.5 });

        document.querySelectorAll('.stat-card').forEach(el => {
            statObserver.observe(el);
        });

        // Track package interest
        document.querySelectorAll('.package-card').forEach(function(card) {
            card.addEventListener('click', function() {
                const packageName = this.querySelector('.package-name').textContent;
                gtag('event', 'package_interest', {
                    'event_category': 'Commercial Security Package',
                    'event_label': packageName,
                    'service_type': 'commercial_security'
                });
            });
        });

        // Track certification badge clicks
        document.querySelectorAll('.compliance-badge, .iso-certification').forEach(function(badge) {
            badge.addEventListener('click', function() {
                gtag('event', 'certification_click', {
                    'event_category': 'Commercial Security Certification',
                    'event_label': this.textContent.trim(),
                    'service_type': 'commercial_security'
                });
            });
        });
    </script>
