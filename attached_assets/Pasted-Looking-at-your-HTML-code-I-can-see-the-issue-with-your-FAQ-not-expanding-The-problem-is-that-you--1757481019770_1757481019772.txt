jQuery Loading Issue
Your jQuery is loaded with defer, but you're trying to use it immediately. Either:

Remove the defer attribute from jQuery:

html<script src="https://code.jquery.com/jquery-3.5.1.slim.min.js"></script>

Or wrap your jQuery code in a document ready that waits for deferred scripts:

javascriptdocument.addEventListener('DOMContentLoaded', function() {
    // Wait a bit more for deferred jQuery to load
    setTimeout(function() {
        if (typeof $ !== 'undefined') {
            // Your existing jQuery FAQ code here
            $('.faq-question').on('click', function() {
                // ... rest of your code
            });
        }
    }, 100);
});
Alternative Pure JavaScript Solution
If you want to avoid jQuery dependency issues, here's a pure JavaScript version:
javascriptdocument.addEventListener('DOMContentLoaded', function() {
    const faqQuestions = document.querySelectorAll('.faq-question');
    
    faqQuestions.forEach(function(question) {
        question.addEventListener('click', function() {
            const item = this.closest('.anaheim-faq-item');
            const toggle = this.querySelector('.faq-toggle i');
            const answer = item.querySelector('.faq-answer');
            const isActive = item.classList.contains('active');
            
            // Close all FAQ items first
            document.querySelectorAll('.anaheim-faq-item').forEach(function(faqItem) {
                faqItem.classList.remove('active');
                const faqToggle = faqItem.querySelector('.faq-toggle i');
                const faqAnswer = faqItem.querySelector('.faq-answer');
                faqToggle.classList.remove('fa-minus');
                faqToggle.classList.add('fa-plus');
                faqAnswer.style.display = 'none';
            });
            
            // If this item wasn't active, open it
            if (!isActive) {
                item.classList.add('active');
                toggle.classList.remove('fa-plus');
                toggle.classList.add('fa-minus');
                answer.style.display = 'block';
            }
        });
    });
});
