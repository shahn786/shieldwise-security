.full-width-section {
    width: 100vw;
    position: relative;
    left: 50%;
    right: 50%;
    margin-left: -50vw;
    margin-right: -50vw;
    background: inherit; /* Keep the section's background */
    box-sizing: border-box;
    overflow-x: hidden;
}
